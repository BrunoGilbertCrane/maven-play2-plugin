81      public static final String ENV_PLAY2_HOME = "PLAY2_HOME";
82  
83      /**
84       * Gets the specified <tt>PLAY2_HOME</tt> location.
85       * This method checks in this order:
86       * <ul>
87       * <li>the PLAY2_HOME system variable</li>
88       * <li>the <tt>play2Home</tt> settings</li>
89       * <li>the PLAY2_HOME environment variable</li>
90       * </ul>
91       * If none is set, this method throws an exception.
92       *
93       * @return the play2 location
94       * @throws MojoExecutionException if the play2 location is not defined.
95       */
96      public String getPlay2HomeOrThrow() throws MojoExecutionException {
97          // First check, system variable
98          String home = System.getProperty(ENV_PLAY2_HOME);
99          if (home != null && !home.isEmpty()) {
100             getLog().debug("Get Play2 home from system variable");
101             return home;
102         }
103 
104         // Second check, the setting configuration
105         if (play2Home != null && !play2Home.isEmpty()) {
106             getLog().debug("Get Play2 home from settings");
107             return play2Home;
108         }
109 
110         // Third check, environment variable
111         home = System.getenv(ENV_PLAY2_HOME);
112         if (home != null && !home.isEmpty()) {
113             getLog().debug("Get Play2 home from environment");
114             return home;
115         }
116 
117         throw new MojoExecutionException(ENV_PLAY2_HOME + " system/configuration/environment variable not set");
118 
119     }

121     public File getPlay2() throws MojoExecutionException {
122         File play2 = null;
123         String path = getPlay2HomeOrThrow();
124         if (isWindows()) {
125             play2 = new File(path, "play.bat");
126         } else {
127             play2 = new File(path, "play");
128         }
129         play2 = manageHomebrew(play2);
130 
131         if (!play2.exists()) {
132             throw new MojoExecutionException("Can't find the play executable in " + path);
133         } else {
134             getLog().debug("Using " + play2.getAbsolutePath());
135         }
136 
137         return play2;
138     }

140     /**
141      * Checks whether the given play executable is in a <tt>Homebrew</tt> managed location.
142      * Homebrew scripts seems to be an issue for play as the provided play executable from this directory is using a
143      *  path expecting relative directories. So, we get such kind of error:
144      * <code>
145      *    /usr/local/Cellar/play/2.0/libexec/play: line 51:
146      *    /usr/local/Cellar/play/2.0/libexec//usr/local/Cellar/play/2.0/libexec/../libexec/framework/build:
147      *    No such file or directory
148      * </code>
149      * In this case we substitute the play executable with the one installed by Homebrew but working correctly.
150      * @param play2 the found play2 executable
151      * @return the given play2 executable except if Homebrew is detected, in this case <tt>/usr/local/bin/play</tt>.
152      */
153     private File manageHomebrew(File play2) {
154         if (play2.getAbsolutePath().startsWith("/usr/local/Cellar/play/")) {
155             getLog().info("Homebrew installation of play detected");
156             // Substitute the play executable by the homebrew one.
157             File file = new File("/usr/local/bin/play");
158             if (! file.exists()) {
159                 getLog().error("Homebrew installation detected, but no play executable in /usr/local/bin");
160             } else {
161                 return file;
162             }
163         }
164         return play2;
165     }
